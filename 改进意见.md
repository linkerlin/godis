## 1. 紧急优先 (P0)

### 1.1 错误处理重构 - 使用 `github.com/cockroachdb/errors`

**问题现状：**
- 项目中使用大量 `panic`，生产环境会导致服务器崩溃
- 错误处理不统一，缺少调用栈信息
- 错误类型混杂（自定义类型、标准 error、字符串）

**改进方案：**

#### 1.1.1 引入错误处理库

```bash
go get github.com/cockroachdb/errors
```

#### 1.1.2 替换 panic 为错误返回

**当前代码（不推荐）：**
```go
// database/server.go
func (server *Server) mustSelectDB(dbIndex int) *DB {
    selectedDB, err := server.selectDB(dbIndex)
    if err != nil {
        panic(err)  // ❌ 生产环境危险
    }
    return selectedDB
}
```

**改进代码：**
```go
// database/server.go
import "github.com/cockroachdb/errors"

func (server *Server) mustSelectDB(dbIndex int) (*DB, error) {
    selectedDB, err := server.selectDB(dbIndex)
    if err != nil {
        return nil, errors.Wrap(err, "select db failed")
    }
    return selectedDB, nil
}

// 调用处处理
func (server *Server) Exec(c redis.Connection, cmdLine [][]byte) (result redis.Reply) {
    defer func() {
        if r := recover(); r != nil {
            //  still catch panic as last resort
            logger.Error(errors.Wrap(errors.Newf("panic: %v", r), "exec command"))
            result = &protocol.UnknownErrReply{}
        }
    }()
    
    // ... 正常逻辑
    selectedDB, err := server.mustSelectDB(dbIndex)
    if err != nil {
        logger.Errorf("%+v", err)  // 打印完整调用栈
        return protocol.MakeErrReply("ERR internal error")
    }
}
```

#### 1.1.3 定义统一的错误类型

```go
// lib/errs/errors.go
package errs

import (
    "github.com/cockroachdb/errors"
)

// 错误代码定义
type ErrorCode int

const (
    ErrCodeUnknown ErrorCode = iota
    ErrCodeWrongType
    ErrCodeProtocol
    ErrCodeKeyNotFound
    ErrCodeInvalidArgs
    ErrCodeInternal
)

// GodisError 统一错误类型
type GodisError struct {
    Code    ErrorCode
    Message string
    Cause   error
}

func (e *GodisError) Error() string {
    return e.Message
}

// 错误构造函数
func NewError(code ErrorCode, msg string) error {
    return errors.WithStack(&GodisError{
        Code:    code,
        Message: msg,
    })
}

func WrapError(err error, code ErrorCode, msg string) error {
    return errors.Wrap(&GodisError{
        Code:    code,
        Message: msg,
        Cause:   err,
    }, msg)
}

// 判断错误类型
func IsWrongType(err error) bool {
    var ge *GodisError
    if errors.As(err, &ge) {
        return ge.Code == ErrCodeWrongType
    }
    return false
}
```

#### 1.1.4 关键路径错误处理改进

```go
// aof/aof.go
import "github.com/cockroachdb/errors"

func (persister *Persister) writeAof(p *payload) error {
    persister.buffer = persister.buffer[:0]
    persister.pausingAof.Lock()
    defer persister.pausingAof.Unlock()
    
    if p.dbIndex != persister.currentDB {
        selectCmd := utils.ToCmdLine("SELECT", strconv.Itoa(p.dbIndex))
        persister.buffer = append(persister.buffer, selectCmd)
        data := protocol.MakeMultiBulkReply(selectCmd).ToBytes()
        _, err := persister.aofFile.Write(data)
        if err != nil {
            // ❌ 原来：logger.Warn(err); return
            // ✅ 现在：包装错误并返回
            return errors.Wrap(err, "write select command to aof failed")
        }
        persister.currentDB = p.dbIndex
    }
    
    data := protocol.MakeMultiBulkReply(p.cmdLine).ToBytes()
    persister.buffer = append(persister.buffer, p.cmdLine)
    _, err := persister.aofFile.Write(data)
    if err != nil {
        return errors.Wrap(err, "write command to aof failed")
    }
    
    // ... 其余逻辑
    return nil
}
```

### 1.3 输入验证与安全加固

```go
// lib/validate/validate.go
package validate

import (
    "github.com/cockroachdb/errors"
)

const (
    MaxKeySize   = 512 * 1024       // 512KB
    MaxValueSize = 512 * 1024 * 1024 // 512MB
    MaxArgCount  = 1000000          // 最大参数数量
)

func ValidateKey(key []byte) error {
    if len(key) == 0 {
        return errors.New("key is empty")
    }
    if len(key) > MaxKeySize {
        return errors.Newf("key too large: %d > %d", len(key), MaxKeySize)
    }
    return nil
}

func ValidateValue(value []byte) error {
    if len(value) > MaxValueSize {
        return errors.Newf("value too large: %d > %d", len(value), MaxValueSize)
    }
    return nil
}
```

### 2.4 锁管理优化

**问题：** 复杂锁顺序可能导致死锁。

**改进：**

```go
// datastruct/dict/concurrent.go
import (
    "context"
    "time"
    "github.com/cockroachdb/errors"
)

const DefaultLockTimeout = 5 * time.Second

func (dict *ConcurrentDict) RWLocksWithTimeout(
    writeKeys []string,
    readKeys []string,
    timeout time.Duration,
) error {
    if timeout == 0 {
        timeout = DefaultLockTimeout
    }
    
    ctx, cancel := context.WithTimeout(context.Background(), timeout)
    defer cancel()
    
    keys := append(writeKeys, readKeys...)
    indices := dict.toLockIndices(keys, false)
    
    for _, index := range indices {
        mu := &dict.table[index].mutex
        select {
        case <-mu.LockChan():  // 需要实现 LockChan
        case <-ctx.Done():
            // 超时，释放已获取的锁
            return errors.New("lock acquire timeout")
        }
    }
    return nil
}
```

### 4.2 优雅关闭

```go
// server.go
func (server *Server) Shutdown(ctx context.Context) error {
    // 1. 停止接受新连接
    server.closing.Set(true)
    
    // 2. 等待活跃请求完成
    done := make(chan struct{})
    go func() {
        server.activeConn.Wait()
        close(done)
    }()
    
    select {
    case <-done:
    case <-ctx.Done():
        return errors.New("shutdown timeout")
    }
    
    // 3. 关闭持久化
    if server.persister != nil {
        if err := server.persister.Close(); err != nil {
            return errors.Wrap(err, "close persister failed")
        }
    }
    
    return nil
}
```
